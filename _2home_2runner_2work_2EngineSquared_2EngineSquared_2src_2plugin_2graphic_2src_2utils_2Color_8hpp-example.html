Converts 8-bit RGB values (0-255) to normalized glm::vec3 (0.0-1.0).Eliminates the need for manual division by 255.0f for each channel, reducing boilerplate and potential for errors.

Parameters
rRed channel (0-255)
gGreen channel (0-255)
bBlue channel (0-255)
Returns
glm::vec3 Normalized color vector with values in range [0.0, 1.0]

auto color = Color::FromRGB(108, 193, 5); // Returns glm::vec3(0.424f, 0.757f, 0.020f)

#pragma once
#include <cstdint>
#include <glm/glm.hpp>
namespace Graphic::Utils {
class Color {
public:
static inline glm::vec3 FromRGB(uint8_t r, uint8_t g, uint8_t b)
{
return glm::vec3(r / 255.f, g / 255.f, b / 255.f);
}
static inline glm::vec3 FromHex(uint32_t hex)
{
uint8_t r = (hex >> 16) & 0xFF;
uint8_t g = (hex >> 8) & 0xFF;
uint8_t b = hex & 0xFF;
return FromRGB(r, g, b);
}
};
} // namespace Graphic::Utils
