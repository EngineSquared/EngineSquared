533 auto &internal = view.get<Component::SoftBodyInternal>(entity);
534
535 if (!internal.IsValid())
536 continue;
537
538 // Get soft body motion properties
539 JPH::BodyLockRead lock(physicsManager.GetPhysicsSystem().GetBodyLockInterface(), internal.bodyID);
540 if (!lock.Succeeded())
541 continue;
542
543 const JPH::Body &body = lock.GetBody();
544 if (!body.IsSoftBody())
545 continue;
546
547 const JPH::SoftBodyMotionProperties *motionProps =
548 static_cast<const JPH::SoftBodyMotionProperties *>(body.GetMotionProperties());
549
550 // Get body center of mass in world space
551 JPH::RVec3 bodyPosition = body.GetCenterOfMassPosition();
552
553 // Update vertex positions (vertices are in local space relative to body center of mass)
554 const auto &joltVertices = motionProps->GetVertices();
555
556 // Write to Mesh.vertices (used by graphics plugin for rendering)
557 auto *mesh = registry.try_get<Object::Component::Mesh>(entity);
558 if (!mesh || mesh->GetVertices().empty())
559 continue;
560
561 // Update Transform with center of mass position (like RigidBody sync)
562 // This way the GPU applies the Transform, and vertices stay in local space
563 auto *transform = registry.try_get<Object::Component::Transform>(entity);
564 if (transform)
565 {
566 transform->SetPosition(Utils::FromJoltRVec3(bodyPosition));
567 // Note: Soft bodies don't have a single rotation, so we leave it as-is
568 }
569
570 // Use the vertex map to sync deduplicated Jolt vertices back to the original mesh
571 // Jolt vertices are in WORLD scale (scaled by initialScale during creation)
572 // We need to convert back to LOCAL mesh space by dividing by initialScale
573 // Then the GPU will apply Transform.scale (which should match initialScale)
574 // Prevent division by zero - use small epsilon for safety
575 constexpr float epsilon = 1e-6f;
576 glm::vec3 safeScale = glm::max(glm::abs(internal.initialScale), glm::vec3(epsilon));
577 // Preserve sign from original scale (treat zero as positive)
578 auto signOrOne = [](float v) { return v < 0.0f ? -1.0f : 1.0f; };
579 safeScale *= glm::vec3(signOrOne(internal.initialScale.x), signOrOne(internal.initialScale.y),
580 signOrOne(internal.initialScale.z));
581 const glm::vec3 invScale = glm::vec3(1.0f) / safeScale;
582
583 if (!internal.vertexMap.empty())
584 {
585 // Use vertex map: original mesh vertices are mapped to deduplicated Jolt vertices
586 for (size_t origIdx = 0u; origIdx < mesh->GetVertices().size() && origIdx < internal.vertexMap.size();
587 ++origIdx)
588 {
589 uint32_t joltIdx = internal.vertexMap[origIdx];
590 if (joltIdx < joltVertices.size())
591 {
592 const auto &v = joltVertices[joltIdx];
593 // Convert from Jolt world-scale space back to mesh local space
594 glm::vec3 worldPos(v.mPosition.GetX(), v.mPosition.GetY(), v.mPosition.GetZ());
595 mesh->SetVertexAt(origIdx, worldPos * invScale);
596 }
597 }
598 }
599 else
600 {
601 // Fallback: direct 1:1 mapping (for procedurally generated meshes)
602 for (size_t i = 0u; i < joltVertices.size() && i < mesh->GetVertices().size(); ++i)
603 {
604 const auto &v = joltVertices[i];
605 // Convert from Jolt world-scale space back to mesh local space
606 glm::vec3 worldPos(v.mPosition.GetX(), v.mPosition.GetY(), v.mPosition.GetZ());
607 mesh->SetVertexAt(i, worldPos * invScale);
608 }
609 }
610
611 // Recalculate normals for correct lighting on deformed soft bodies
612 Object::Utils::RecalculateNormals(*mesh);
613 }
